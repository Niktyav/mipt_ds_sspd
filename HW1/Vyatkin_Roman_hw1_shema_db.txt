Table Transactions {
  id integer [primary key]
  product_id integer
  customer_id integer
  transaction_date datetime 
  online_order boolean
  order_status_id integer
}

Table Products {
  id integer [primary key]
  brand_id integer
  product_line_id integer
  product_class_id integer
  product_size_id integer
  list_price float
  standard_cost float
}

Table Customers {
  id integer [primary key]
  first_name text
  last_name text
  gender varchar
  DOB varchar
  job_title text
  job_industry_category_id integer
  wealth_segment_id integer
  deceased_indicator bool
  owns_car bool
  property_valuation integer
  address varchar
  postcode integer
  state_id integer
  country_id integer
}
